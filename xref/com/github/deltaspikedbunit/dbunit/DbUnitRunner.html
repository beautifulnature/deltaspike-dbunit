57      /**
58       * Extracts a DBUnit database operation from a given configuration and internal database operation.
59       *
60       * @param dbUnitConfiguration current configuration
61       * @param databaseOperation internal database operation
62       * @return a DBUnit database operation. If no internal operation is selected,
63       *  a CLEAN_INSERT default operation is returned
64       */
65      protected org.dbunit.operation.DatabaseOperation dbUnitOperation(DbUnitConfiguration dbUnitConfiguration, DatabaseOperation databaseOperation) {
66          if (databaseOperation != null) {
67              Class<? extends DatabaseOperationLookup> databaseOperationLookupClass = DefaultDatabaseOperationLookup.class;
68              if (dbUnitConfiguration != null) {
69                  databaseOperationLookupClass = dbUnitConfiguration.databaseOperationLookup();
70              }
71              DatabaseOperationLookup databaseOperationLookup;
72              try {
73                  databaseOperationLookup = databaseOperationLookupClass.newInstance();
74              } catch (Exception e) {
75                  throw new IllegalArgumentException("Unable to create database operation lookup instance for "
76                          + databaseOperationLookupClass, e);
77              }
78              return databaseOperationLookup.get(databaseOperation);
79          }
80          return org.dbunit.operation.DatabaseOperation.CLEAN_INSERT;
81      }

83      /**
84       * Selects a dataset loader, given a configuration.
85       *
86       * @param dbUnitConfiguration
87       * @return a dataset loader. Default is {@link FlatXmlDataSetLoader} if no configuration available
88       */
89      protected DataSetLoader dataSetLoader(DbUnitConfiguration dbUnitConfiguration) {
90          if (dbUnitConfiguration == null || dbUnitConfiguration.dataSetLoader() == null) {
91              return new FlatXmlDataSetLoader();
92          }
93          try {
94              return dbUnitConfiguration.dataSetLoader().newInstance();
95          } catch (InstantiationException | IllegalAccessException e) {
96              throw new IllegalStateException(e);
97          }
98      }

122     /**
123      * Loads all given datasets relative to their test class location using the selected loader.
124      *
125      * @param loader
126      * @param dataSetLocations
127      * @param resourceBase
128      * @return the datasets list
129      */
130     private List<IDataSet> loadDataSets(DataSetLoader loader, String[] dataSetLocations, Class<?> resourceBase) {
131         List<IDataSet> datasets = new ArrayList<>();
132         if (dataSetLocations != null) {
133             for (String value : dataSetLocations) {
134                 if (value != null) {
135                     IDataSet ds = null;
136                     try {
137                         ds = loader.loadDataSet(resourceBase, value);
138                     } catch (Exception e) {
139                         throw new RuntimeException(e);
140                     }
141                     datasets.add(ds);
142                 }
143             }
144         }
145         return datasets;
146     }

148     /**
149      * Verifies expected db content.
150      *
151      * @param testContext current test context to lookup class and method level annotations
152      * @throws Exception
153      */
154     public void verifyExpected(DbUnitTestContext testContext) throws Exception {
155         List<com.github.deltaspikedbunit.annotation.ExpectedDatabase> expectedDatabases = new ArrayList<>();
156         com.github.deltaspikedbunit.annotation.ExpectedDatabase classExpectedDb = testContext.getTestClass().getAnnotation(com.github.deltaspikedbunit.annotation.ExpectedDatabase.class);
157         if (classExpectedDb != null) {
158             expectedDatabases.add(classExpectedDb);
159         }
160         com.github.deltaspikedbunit.annotation.ExpectedDatabase expectedDb = testContext.getTestMethod().getAnnotation(com.github.deltaspikedbunit.annotation.ExpectedDatabase.class);
161         if (expectedDb != null) {
162             expectedDatabases.add(expectedDb);
163         }
164         verifyExpected(testContext, expectedDatabases);
165     }

167     /**
168      * Loads the given dataset relative to its test class location using the selected modifier.
169      *
170      * @param testContext
171      * @param dataSetLocation
172      * @param modifier
173      * @return
174      * @throws Exception
175      */
176     private IDataSet loadDataset(DbUnitTestContext testContext, String dataSetLocation, DataSetModifier modifier)
177             throws Exception {
178         DataSetLoader dataSetLoader = testContext.getDataSetLoader();
179         if (dataSetLocation != null && dataSetLocation.length() > 0) {
180             IDataSet dataSet = dataSetLoader.loadDataSet(testContext.getTestClass(), dataSetLocation);
181             dataSet = modifier.modify(dataSet);
182             if (dataSet == null) {
183                 throw new IllegalArgumentException("Unable to load dataset from \"" + dataSetLocation + "\" using " + dataSetLoader.getClass());
184             }
185             return dataSet;
186         }
187         return null;
188     }

190     /**
191      * Verifies expected db content. If an exception is found in the test context, from one of the
192      * preceding stages, verification is skipped.
193      *
194      * @param testContext current test context
195      * @param annotations expected database annotations. These are verified from last to first and
196      *  all the following checks are skipped once an <quote>override</quote> value of true is found.
197      * @throws Exception
198      */
199     protected void verifyExpected(DbUnitTestContext testContext, List<com.github.deltaspikedbunit.annotation.ExpectedDatabase> annotations) throws Exception {
200         if (testContext.getTestException() != null) {
201             if (LOG.isLoggable(Level.INFO)) {
202                 LOG.info("Skipping @DatabaseTest expectation due to test exception "
203                         + testContext.getTestException().getClass());
204             }
205             return;
206         }
207 //        DatabaseConnections connections = testContext.getConnections();
208         DataSetModifier modifier = getModifier(testContext, annotations);
209 //        DataSetModifiers modifiers = new DataSetModifiers();
210 //        for (Class<? extends DataSetModifier> modifierClass : annotation.modifiers()) {
211 //            modifiers.add(testContext.getTestInstance(), modifierClass);
212 //        }
213 
214         for (int i = annotations.size() - 1; i >= 0; i--) {
215             com.github.deltaspikedbunit.annotation.ExpectedDatabase annotation = annotations.get(i);
216             String query = annotation.query();
217             String table = annotation.table();
218             IDataSet expectedDataSet = loadDataset(testContext, annotation.value(), modifier);
219             if (expectedDataSet != null) {
220                 if (LOG.isLoggable(Level.FINE)) {
221                     LOG.info("Verifying @DatabaseTest expectation using " + annotation.value());
222                 }
223                 DatabaseAssertion assertion = annotation.assertionMode().getDatabaseAssertion();
224                 if (query != null && query.length() > 0) {
225                     if ( table == null || table.length() == 0 ) {
226                         throw new IllegalArgumentException("The table name must be specified when using a SQL query");
227                     }
228                     ITable expectedTable = expectedDataSet.getTable(table);
229                     ITable actualTable = dbUnitConnection.createQueryTable(table, query);
230                     assertion.assertEquals(expectedTable, actualTable);
231                 } else if (table != null && table.length() > 0) {
232                     ITable actualTable = dbUnitConnection.createTable(table);
233                     ITable expectedTable = expectedDataSet.getTable(table);
234                     assertion.assertEquals(expectedTable, actualTable);
235                 } else {
236                     IDataSet actualDataSet = dbUnitConnection.createDataSet();
237                     assertion.assertEquals(expectedDataSet, actualDataSet);
238                 }
239             }
240             if (annotation.override()) {
241                 // No need to test any more
242                 return;
243             }
244         }
245     }

247     /**
248      * Extracts all modifiers given a test context and a list of expectations.
249      *
250      * @param testContext
251      * @param annotations
252      * @return
253      */
254     private DataSetModifier getModifier(DbUnitTestContext testContext, List<com.github.deltaspikedbunit.annotation.ExpectedDatabase> annotations) {
255         DataSetModifiers modifiers = new DataSetModifiers();
256         for (com.github.deltaspikedbunit.annotation.ExpectedDatabase annotation : annotations) {
257             for (Class<? extends DataSetModifier> modifierClass : annotation.modifiers()) {
258                 modifiers.add(testContext.getTestInstance(), modifierClass);
259             }
260         }
261         return modifiers;
262     }

264     /**
265      * Adds all declared database setup related datasets to a DBUnit database tester.
266      *
267      * @param testContext the current test context
268      * @param dbUnitConfiguration the current test configuration
269      * @param databaseTester the DBUnit database tester
270      */
271     public void prepareTesterSetup(DbUnitTestContext testContext, DbUnitConfiguration dbUnitConfiguration, IDatabaseTester databaseTester) {
272         DataSetLoader dataSetLoader = dataSetLoader(dbUnitConfiguration);
273         testContext.setDataSetLoader(dataSetLoader);
274         List<IDataSet> dataSets = new ArrayList<>();
275         DatabaseSetup classDbSetup = testContext.getTestClass().getAnnotation(DatabaseSetup.class);
276         databaseTester.setSetUpOperation(org.dbunit.operation.DatabaseOperation.NONE);
277         if (classDbSetup != null) {
278             dataSets.addAll(loadDataSets(testContext, classDbSetup));
279             databaseTester.setSetUpOperation(dbUnitOperation(dbUnitConfiguration, classDbSetup.type()));
280         }
281         DatabaseSetup dbSetup = testContext.getTestMethod().getAnnotation(DatabaseSetup.class);
282         if (dbSetup != null) {
283             dataSets.addAll(loadDataSets(testContext, dbSetup));
284             databaseTester.setSetUpOperation(dbUnitOperation(dbUnitConfiguration, dbSetup.type()));
285         }
286         if (dataSets != null && !dataSets.isEmpty()) {
287             IDataSet ds = null;
288             try {
289                 ds = new CompositeDataSet(dataSets.toArray(new IDataSet[dataSets.size()]));
290             } catch (DataSetException e) {
291                 throw new RuntimeException(e);
292             }
293             databaseTester.setDataSet(ds);
294         }
295     }

297     /**
298      * Adds all declared database tear down related datasets to a DBUnit database tester.
299      *
300      * @param testContext the current test context
301      * @param dbUnitConfiguration the current test configuration
302      * @param databaseTester the DBUnit database tester
303      */
304     public void prepareTesterTearDown(DbUnitTestContext testContext, DbUnitConfiguration dbUnitConfiguration, IDatabaseTester databaseTester) {
305         DataSetLoader tearDownDataSetLoader = dataSetLoader(dbUnitConfiguration);
306         testContext.setDataSetLoader(tearDownDataSetLoader);
307         List<IDataSet> tearDownDataSets = new ArrayList<>();
308         DatabaseTearDown classDbTearDown = testContext.getTestClass().getAnnotation(DatabaseTearDown.class);
309         if (classDbTearDown != null) {
310             tearDownDataSets.addAll(loadDataSets(testContext, classDbTearDown));
311             databaseTester.setTearDownOperation(dbUnitOperation(dbUnitConfiguration, classDbTearDown.type()));
312         }
313         DatabaseTearDown dbTearDown = testContext.getTestMethod().getAnnotation(DatabaseTearDown.class);
314         if (dbTearDown != null) {
315             tearDownDataSets.addAll(loadDataSets(testContext, dbTearDown));
316             databaseTester.setTearDownOperation(dbUnitOperation(dbUnitConfiguration, dbTearDown.type()));
317         }
318         if (tearDownDataSets != null && !tearDownDataSets.isEmpty()) {
319             IDataSet ds = null;
320             try {
321                 ds = new CompositeDataSet(tearDownDataSets.toArray(new IDataSet[tearDownDataSets.size()]));
322             } catch (DataSetException e) {
323                 throw new RuntimeException(e);
324             }
325             databaseTester.setDataSet(ds);
326         }
327     }
